# 常用命令：
    ## 获取环境变量
            echo %path%

    ## 查看网络连通性：
            ping -t www.baidu.com

    ## 查看任务：
    		tasklist /?
       
    ## 终止任务：	
    		taskkill /?        

    ## telnet
    
    	cmd> telnet localhost 8080

    	cmd> Ctrl+]

    	Telnet>	send hello world	
    	Telnet>	?/h
